203 /* SHPAPI_CALL */
204 /* */
205 /* The following two macros are present to allow forcing */
206 /* various calling conventions on the Shapelib API. */
207 /* */
208 /* To force __stdcall conventions (needed to call Shapelib */
209 /* from Visual Basic and/or Dephi I believe) the makefile could */
210 /* be modified to define: */
211 /* */
212 /* /DSHPAPI_CALL=__stdcall */
213 /* */
214 /* If it is desired to force export of the Shapelib API without */
215 /* using the shapelib.def file, use the following definition. */
216 /* */
217 /* /DSHAPELIB_DLLEXPORT */
218 /* */
219 /* To get both at once it will be necessary to hack this */
220 /* include file to define: */
221 /* */
222 /* #define SHPAPI_CALL __declspec(dllexport) __stdcall */
223 /* #define SHPAPI_CALL1 __declspec(dllexport) * __stdcall */
224 /* */
225 /* The complexity of the situation is partly caused by the */
226 /* peculiar requirement of Visual C++ that __stdcall appear */
227 /* after any "*"'s in the return value of a function while the */
228 /* __declspec(dllexport) must appear before them. */
229 /* -------------------------------------------------------------------- */
230 
231 #ifdef SHAPELIB_DLLEXPORT
232 # define SHPAPI_CALL __declspec(dllexport)
233 # define SHPAPI_CALL1(x) __declspec(dllexport) x
234 #endif
