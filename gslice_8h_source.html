60 /* we go through extra hoops to ensure type safety */
61 #define g_slice_dup(type, mem) \
62  (1 ? (type*) g_slice_copy (sizeof (type), (mem)) \
63  : ((void) ((type*) 0 == (mem)), (type*) 0))
64 #define g_slice_free(type, mem) do { \
65  if (1) g_slice_free1 (sizeof (type), (mem)); \
66  else (void) ((type*) 0 == (mem)); \
67 } while (0)
68 #define g_slice_free_chain(type, mem_chain, next) do { \
69  if (1) g_slice_free_chain_with_offset (sizeof (type), \
70  (mem_chain), G_STRUCT_OFFSET (type, next)); \
71  else (void) ((type*) 0 == (mem_chain)); \
72 } while (0)
