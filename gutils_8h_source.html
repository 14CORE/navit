76 /* inlining hassle. for compilers that don't allow the `inline' keyword,
77  * mostly because of strict ANSI C compliance or dumbness, we try to fall
78  * back to either `__inline__' or `__inline'.
79  * G_CAN_INLINE is defined in glibconfig.h if the compiler seems to be
80  * actually *capable* to do function inlining, in which case inline
81  * function bodies do make sense. we also define G_INLINE_FUNC to properly
82  * export the function prototypes if no inlining can be performed.
83  * inline function bodies have to be special cased with G_CAN_INLINE and a
84  * .c file specific macro to allow one compiled instance with extern linkage
85  * of the functions by defining G_IMPLEMENT_INLINES and the .c file macro.
86  */

325 G_INLINE_FUNC gint
326 g_bit_nth_lsf (gulong mask,
327  gint nth_bit)
328 {
329  if (G_UNLIKELY (nth_bit < -1))
330  nth_bit = -1;
331  while (nth_bit < ((GLIB_SIZEOF_LONG * 8) - 1))
332  {
333  nth_bit++;
334  if (mask & (1UL << nth_bit))
335  return nth_bit;
336  }
337  return -1;
338 }
339 G_INLINE_FUNC gint
340 g_bit_nth_msf (gulong mask,
341  gint nth_bit)
342 {
343  if (nth_bit < 0 || G_UNLIKELY (nth_bit > GLIB_SIZEOF_LONG * 8))
344  nth_bit = GLIB_SIZEOF_LONG * 8;
345  while (nth_bit > 0)
346  {
347  nth_bit--;
348  if (mask & (1UL << nth_bit))
349  return nth_bit;
350  }
351  return -1;
352 }
353 G_INLINE_FUNC guint
354 g_bit_storage (gulong number)
355 {
356 #if defined(__GNUC__) && (__GNUC__ >= 4) && defined(__OPTIMIZE__) && !defined(__ARM_ARCH_4T__)
357  return G_LIKELY (number) ?
358  ((GLIB_SIZEOF_LONG * 8 - 1) ^ __builtin_clzl(number)) + 1 : 1;
359 #else
360  register guint n_bits = 0;
361 
362  do
363  {
364  n_bits++;
365  number >>= 1;
366  }
367  while (number);
368  return n_bits;
369 #endif
370 }
